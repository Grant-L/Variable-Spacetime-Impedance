\begin{equation}
    \frac{dN}{dt} = R_{g} N(t)
\end{equation}
Where $R_g$ is the \textbf{Node Genesis Rate} (Hz). Solving for $N(t)$:
\begin{equation}
    N(t) = N_0 e^{R_g t}
\end{equation}

\begin{equation}
    v = \frac{dD}{dt} = l_0 \frac{dN}{dt} = l_0 (R_g N) = R_g D

\begin{equation}
