\section{Exercises}
\label{sec:ch5_exercises}

\begin{enumerate}
    \item \textbf{The Range Limit:} 
    \citestart Using the saturation frequency $\omega_{sat}$ from Chapter 1, show that the interaction range $D$ for a $W$-frequency signal is $\approx 10^{-18}$ meters[cite: 1109].

    \item \textbf{Impedance Mismatch:} 
    Model Beta decay as a signal traveling from a high-impedance saturated node to a low-impedance ground-state node. \citestart Calculate the reflection coefficient $\Gamma$[cite: 1110, 1111].

    \item \textbf{Bias Coupling:} 
    \citestart If the vacuum orientation $\mathbf{\Omega}_{vac}$ were to shift by $5^{\circ}$, calculate the resulting change in the observed Weinberg Angle $\theta_W$[cite: 1111].
\end{enumerate}
