\section{Exercises}
\label{sec:ch6_exercises}

\begin{enumerate}
    \item \textbf{The Redshift Correction:} 
    \citestart Derive the relationship between cosmological redshift $z$ and the shifting impedance $Z_0(t)$[cite: 1151].

    \item \textbf{High-Flux Biology:} 
    \citestart Calculate the required $Z_0$ value in a low-impedance epoch that would allow biological structures to maintain double their modern skeletal stress limit[cite: 1152].

    \item \textbf{Quench Rate:} 
    \citestart Given the measured stability of $c$ over the last 100 years, calculate the upper bound for the modern Quench Constant $\gamma$[cite: 1153].
\end{enumerate}
