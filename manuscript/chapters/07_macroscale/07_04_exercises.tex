\section{Exercises}
\label{sec:ch7_exercises}

\begin{enumerate}
    \item \textbf{Rotation Flattening:} 
    \citestart Given a galactic core mass $M$, find the radius $r$ where the radial impedance gradient exactly balances the Newtonian $1/r^2$ decay to produce a flat velocity profile[cite: 421].

    \item \textbf{Lattice Relaxation:} 
    \citestart Calculate the time $\tau$ required for a region of metric strain to return to $Z_0$ after a mass displacement, using the Quench Constant $\gamma$ from Chapter 6[cite: 422].

    \item \textbf{Refractive Lensing:} 
    \citestart Show that the bending of light in a radial impedance gradient $Z(r)$ recovers the Einstein deflection angle $\theta = 4GM/rc^2$ without invoking four-dimensional geometric curvature[cite: 423].
\end{enumerate}
