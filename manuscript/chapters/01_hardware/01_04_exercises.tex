\section{Exercises}
\label{sec:ch1_exercises}

\begin{enumerate}
    \item \textbf{Deriving the Slew Rate:}
    Using the defined hardware constants $L_{node} \approx 1.26 \times 10^{-6} \text{ H/m}$ and $C_{node} \approx 8.85 \times 10^{-12} \text{ F/m}$, calculate the Global Slew Rate ($c$) and compare it to the standard speed of light. What is the percentage error, if any?

    \item \textbf{The Saturation Limit:}
    Assume a Lattice Pitch of $l_P \approx 1.62 \times 10^{-35} \text{ m}$. Calculate the \textbf{Saturation Frequency} $\omega_{sat}$ (Eq. \ref{eq:omega_sat}). 
    \textit{Hint: This should correspond to the inverse of the Planck Time.}

    \item \textbf{Chiral Impedance Calculation:}
    A signal with helicity vector $\mathbf{J}$ propagates antiparallel to the vacuum orientation $\mathbf{\Omega}_{vac}$ ($\theta = 180^{\circ}$). 
    If the Asymmetry Coefficient $\eta = 1.0$, calculate the effective impedance $Z_{metric}$ experienced by this signal.
    \textit{Discussion: How does this result explain the non-propagation of Right-Handed neutrinos?}
\end{enumerate}
