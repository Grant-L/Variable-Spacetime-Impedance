\section{Exercises}
\label{sec:ch8_exercises}

\begin{enumerate}
    \item \textbf{Refractive Index Calculation:} 
    \citestart Find the Lattice Stress $\sigma$ required to achieve an effective velocity of $2c$ relative to a stationary observer[cite: 457].

    \item \textbf{Tapering Geometry:} 
    \citestart Design an impedance gradient profile that minimizes reflective loss (Cherenkov emission) at a bubble boundary traveling at $0.9c$[cite: 459].

    \item \textbf{Short-Circuit Power:} 
    \citestart Using the hardware constants from Chapter 1, estimate the Joules per cubic micron yielded by a topological short in a ground-state vacuum where $Z_{0} = 376.73\,\Omega$[cite: 460].
\end{enumerate}
